Создание и наполнение таблицы получателей обратной связи:

create table Recipients(
    RECIPIENT_ID SERIAL,
    RECIPIENT_FULL_NAME VARCHAR,
    RECIPIENT_EMAIL VARCHAR,
    PRIMARY KEY(RECIPIENT_ID)
);
